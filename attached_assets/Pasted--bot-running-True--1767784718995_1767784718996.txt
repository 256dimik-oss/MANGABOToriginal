# ---------- Глобальные переменные ----------
bot_running = True  # для управления запуском/остановкой бота
hint_text = ""      # текст-подсказка для вкладки Add Idea

# ---------- Загрузка подсказки из Excel ----------
def load_hint():
    global hint_text
    wb = get_book()
    if "Admin" not in wb.sheetnames:
        ws_admin = wb.create_sheet("Admin")
        ws_admin["A1"] = "Hint text"
        ws_admin["A2"] = ""
        wb.save(FILE)
    else:
        ws_admin = wb["Admin"]
    hint_text = ws_admin["A2"].value or ""

# ---------- Сохранение подсказки в Excel ----------
def save_hint(new_hint):
    global hint_text
    wb = get_book()
    ws_admin = wb["Admin"]
    ws_admin["A2"].value = new_hint
    wb.save(FILE)
    hint_text = new_hint

# ---------- Остановка и запуск бота ----------
@bot.message_handler(commands=["stop"])
def stop_bot(msg):
    global bot_running
    if msg.from_user.id != ADMIN_ID:
        return
    bot_running = False
    bot.reply_to(msg, "⛔ Бот остановлен.")

@bot.message_handler(commands=["start_bot"])
def start_bot(msg):
    global bot_running
    if msg.from_user.id != ADMIN_ID:
        return
    bot_running = True
    bot.reply_to(msg, "✅ Бот снова запущен.")

# ---------- Добавление идеи с подсказкой ----------
@bot.message_handler(func=lambda m: m.text == "➕ Добавить идею")
def add_idea(msg):
    if not bot_running:
        bot.send_message(msg.chat.id, "⛔ Бот сейчас остановлен. Попробуй позже.")
        return

    # Проверка: пользователь должен оценить все новые идеи
    wb = get_book()
    ws = wb.active
    total_ideas = [row[0] for row in ws.iter_rows(min_row=2, values_only=True) if row[2] != str(msg.from_user.id)]
    seen_count = len(user_seen.get(msg.from_user.id, []))
    if seen_count < len(total_ideas):
        bot.send_message(msg.chat.id, f"⚠️ Сначала оцени все новые идеи! Осталось {len(total_ideas) - seen_count}")
        return

    # Загружаем подсказку
    load_hint()
    if hint_text:
        bot.send_message(msg.chat.id, f"💡 Текущая часть истории: {hint_text}")

    bot.send_message(msg.chat.id, "✍️ Напиши свою идею одним сообщением:")

# ---------- Изменение подсказки админом ----------
@bot.message_handler(commands=["hint"])
def change_hint(msg):
    if msg.from_user.id != ADMIN_ID:
        return
    # Команда: /hint новый текст
    new_hint = msg.text[len("/hint "):].strip()
    save_hint(new_hint)
    bot.reply_to(msg, "✅ Подсказка обновлена.")
